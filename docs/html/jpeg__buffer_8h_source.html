92 static boolean fill_minput_buffer(j_decompress_ptr cinfo) {
93  my_src_mptr src = (my_src_mptr) cinfo->src;
94  size_t nbytes;
95  if (src->indexinmem + INPUT_BUF_SIZE>src->lenght) nbytes=src->lenght - src->indexinmem;
96  else nbytes = INPUT_BUF_SIZE;
97  std::memcpy(src->buffer,src->inmem,nbytes);
98  src->inmem += nbytes;
99  src->indexinmem += (int)nbytes;
100  src->pub.next_input_byte = src->buffer;
101  src->pub.bytes_in_buffer = INPUT_BUF_SIZE;
102  src->start_of_file = FALSE;
103  return TRUE;
104 }

106 static void skip_minput_data(j_decompress_ptr cinfo, long num_bytes) {
107  my_src_ptr src = (my_src_ptr)cinfo->src;
108  if (num_bytes > 0) {
109  while (num_bytes > (long) src->pub.bytes_in_buffer) {
110  num_bytes -= (long) src->pub.bytes_in_buffer;
111  fill_minput_buffer(cinfo);
112  // note we assume that fill_input_buffer will never return FALSE,
113  // so suspension need not be handled.
114  //
115  }
116  src->pub.next_input_byte += (size_t) num_bytes;
117  src->pub.bytes_in_buffer -= (size_t) num_bytes;
118  }
119 }

130 static void jpeg_mem_src(j_decompress_ptr cinfo, JOCTET * memptr,int lenght) {
131  my_src_mptr src;
132 
133  // The source object and input buffer are made permanent so that a series
134  //of JPEG images can be read from the same file by calling jpeg_stdio_src
135  // only before the first one. (If we discarded the buffer at the end of
136  // one image, we'd likely lose the start of the next one.)
137  // This makes it unsafe to use this manager and a different source
138  // manager serially with the same JPEG object. Caveat programmer.
139  //
140 
141  // first time for this JPEG object?
142  if (cinfo->src == NULL) {
143  cinfo->src = (struct jpeg_source_mgr*)(*cinfo->mem->alloc_small)((j_common_ptr) cinfo,
144  JPOOL_PERMANENT,sizeof(my_source_mem));
145  src = (my_src_mptr) cinfo->src;
146  src->buffer = (JOCTET *)(*cinfo->mem->alloc_small) ((j_common_ptr) cinfo,
147  JPOOL_PERMANENT,INPUT_BUF_SIZE * sizeof(JOCTET));
148  }
149 
150  src = (my_src_mptr) cinfo->src;
151  src->pub.init_source = init_msource;
152  src->pub.fill_input_buffer = fill_minput_buffer;
153  src->pub.skip_input_data = skip_minput_data;
154  //src->pub.resync_to_restart = jpeg_resync_to_restart; // use default method
155  src->pub.term_source = term_source;
156  src->inmem = memptr;
157  src->indexinmem = 0;
158  src->lenght = lenght;
159  src->pub.bytes_in_buffer = 0; // forces fill_input_buffer on first read
160  src->pub.next_input_byte = NULL; // until buffer loaded
161 }

227 static CImg get_load_jpeg_buffer(const JOCTET *const buffer, const unsigned buffer_size) {
228  struct jpeg_decompress_struct cinfo;
229  struct jpeg_error_mgr jerr;
230  cinfo.err = jpeg_std_error(&jerr);
231  jpeg_create_decompress(&cinfo);
232  jpeg_mem_src(&cinfo,const_cast<JOCTET*>(buffer),buffer_size);
233  jpeg_read_header(&cinfo,TRUE);
234  jpeg_start_decompress(&cinfo);
235 
236  const unsigned int row_stride = cinfo.output_width * cinfo.output_components;
237  JOCTET *buf = new JOCTET[cinfo.output_width*cinfo.output_height*cinfo.output_components];
238  const JOCTET *buf2 = buf;
239  JSAMPROW row_pointer[1];
240  while (cinfo.output_scanline < cinfo.output_height) {
241  row_pointer[0] = buf + cinfo.output_scanline*row_stride;
242  jpeg_read_scanlines(&cinfo,row_pointer,1);
243  }
244  jpeg_finish_decompress(&cinfo);
245  jpeg_destroy_decompress(&cinfo);
246 
247  CImg<T> dest(cinfo.output_width,cinfo.output_height,1,cinfo.output_components);
248  switch (dest.spectrum()) {
249  case 1: {
250  T *ptr_g = dest.data(0,0,0,0);
251  cimg_foroff(dest,off) *(ptr_g++) = (T)*(buf2++);
252  } break;
253  case 3: {
254  T
255  *ptr_r = dest.data(0,0,0,0),
256  *ptr_g = dest.data(0,0,0,1),
257  *ptr_b = dest.data(0,0,0,2);
258  cimg_forXY(dest,x,y) {
259  *(ptr_r++) = (T)*(buf2++);
260  *(ptr_g++) = (T)*(buf2++);
261  *(ptr_b++) = (T)*(buf2++);
262  }
263  } break;
264  case 4: {
265  T
266  *ptr_r = dest.data(0,0,0,0),
267  *ptr_g = dest.data(0,0,0,1),
268  *ptr_b = dest.data(0,0,0,2),
269  *ptr_a = dest.data(0,0,0,3);
270  cimg_forXY(dest,x,y) {
271  *(ptr_r++) = (T)*(buf2++);
272  *(ptr_g++) = (T)*(buf2++);
273  *(ptr_b++) = (T)*(buf2++);
274  *(ptr_a++) = (T)*(buf2++);
275  }
276  } break;
277  }
278  delete[] buf;
279 
280  return dest;
281 }

299 const CImg& save_jpeg_buffer(JOCTET *const buffer, unsigned int &buffer_size, const int quality=100) const {
300 
301  // Fill pixel buffer
302  JOCTET *buf;
303  unsigned int dimbuf=0;
304  J_COLOR_SPACE colortype=JCS_RGB;
305  switch (spectrum()) {
306  case 1: {
307  // Greyscale images
308  JOCTET *buf2 = buf = new JOCTET[width()*height()*(dimbuf=1)];
309  const T
310  *ptr_g = data();
311  colortype = JCS_GRAYSCALE;
312  cimg_foroff(*this,off) *(buf2++) = (JOCTET)*(ptr_g++);
313  } break;
314  case 2:
315  case 3: {
316  // RGB images
317  JOCTET *buf2 = buf = new JOCTET[width()*height()*(dimbuf=3)];
318  const T
319  *ptr_r = data(0,0,0,0),
320  *ptr_g = data(0,0,0,1),
321  *ptr_b = data(0,0,0,spectrum()>2?2:0);
322  colortype = JCS_RGB;
323  cimg_forXY(*this,x,y) {
324  *(buf2++) = (JOCTET)*(ptr_r++);
325  *(buf2++) = (JOCTET)*(ptr_g++);
326  *(buf2++) = (JOCTET)*(ptr_b++);
327  }
328  } break;
329  default: {
330  // YCMYK images
331  JOCTET *buf2 = buf = new JOCTET[width()*height()*(dimbuf=4)];
332  const T
333  *ptr_r = data(0,0,0,0),
334  *ptr_g = data(0,0,0,1),
335  *ptr_b = data(0,0,0,2),
336  *ptr_a = data(0,0,0,3);
337  colortype = JCS_CMYK;
338  cimg_forXY(*this,x,y) {
339  *(buf2++) = (JOCTET)*(ptr_r++);
340  *(buf2++) = (JOCTET)*(ptr_g++);
341  *(buf2++) = (JOCTET)*(ptr_b++);
342  *(buf2++) = (JOCTET)*(ptr_a++);
343  }
344  } break;
345  }
346 
347  // Call libjpeg functions
348  struct jpeg_compress_struct cinfo;
349  struct jpeg_error_mgr jerr;
350  cinfo.err = jpeg_std_error(&jerr);
351  jpeg_create_compress(&cinfo);
352  jpeg_mem_dest(&cinfo, buffer, buffer_size);
353  cinfo.image_width = width();
354  cinfo.image_height = height();
355  cinfo.input_components = dimbuf;
356  cinfo.in_color_space = colortype;
357  jpeg_set_defaults(&cinfo);
358  jpeg_set_quality(&cinfo,quality<100?quality:100,TRUE);
359  jpeg_start_compress(&cinfo,TRUE);
360 
361  const unsigned int row_stride = width()*dimbuf;
362  JSAMPROW row_pointer[1];
363  while (cinfo.next_scanline < cinfo.image_height) {
364  row_pointer[0] = &buf[cinfo.next_scanline*row_stride];
365  jpeg_write_scanlines(&cinfo,row_pointer,1);
366  }
367  jpeg_finish_compress(&cinfo);
368  delete[] buf;
369  buffer_size = jpeg_mem_size(&cinfo);
370  jpeg_destroy_compress(&cinfo);
371  return *this;
372 }
