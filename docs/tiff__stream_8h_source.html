126 CImg<T>& load_tiff(std::istream* tiffInStream,
127  const unsigned int first_frame=0, const unsigned int last_frame=~0U,
128  const unsigned int step_frame=1)
129 {
130  const unsigned int
131  nfirst_frame = first_frame<last_frame?first_frame:last_frame,
132  nstep_frame = step_frame?step_frame:1;
133  unsigned int nlast_frame = first_frame<last_frame?last_frame:first_frame;
134 
135  TIFF *tif = TIFFStreamOpen("MemTiff", tiffInStream);
136  if (tif)
137  {
138  unsigned int nb_images = 0;
139  do {
140  ++nb_images;
141  } while (TIFFReadDirectory(tif));
142 
143  if (nfirst_frame>=nb_images || (nlast_frame!=~0U && nlast_frame>=nb_images))
144  {
145  cimg::warn("load_tiff(): Invalid specified frame range is [%u,%u] (step %u) "
146  "since stream contains %u image(s).",
147  nfirst_frame,nlast_frame,nstep_frame,nb_images);
148  }
149 
150  if (nfirst_frame>=nb_images)
151  {
152  return assign();
153  }
154 
155  if (nlast_frame>=nb_images)
156  {
157  nlast_frame = nb_images - 1;
158  }
159  TIFFSetDirectory(tif,0);
160  CImg<T> frame;
161  for (unsigned int l = nfirst_frame; l<=nlast_frame; l+=nstep_frame) {
162  frame._load_tiff(tif,l,0,0);
163  if (l==nfirst_frame)
164  assign(frame._width,frame._height,1 + (nlast_frame - nfirst_frame)/nstep_frame,frame._spectrum);
165  if (frame._width>_width || frame._height>_height || frame._spectrum>_spectrum)
166  resize(std::max(frame._width,_width),std::max(frame._height,_height),
167  -100,std::max(frame._spectrum,_spectrum),0);
168  draw_image(0,0,(l - nfirst_frame)/nstep_frame,frame);
169  }
170  TIFFClose(tif);
171  }
172  else
173  {
174  throw CImgIOException(_cimg_instance
175  "load_tiff(): Failed to read data from stream",
176  cimg_instance);
177  }
178 
179  return *this;
180 }
